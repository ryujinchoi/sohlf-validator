\begin{abstract}
This document establishes the rigorous mathematical foundations for mitigating hallucination phenomena in autoregressive Large Language Models (LLMs). By mapping token logit uncertainty to Shannon Information Entropy and bounding hidden state vectors to a canonical Contextual Manifold via localized geometric projections, we mathematically prove the existence of a zero-distortion alignment pathway that preserves semantic truth values during text generation.
\end{abstract}
\section{Theorem 1: Token Uncertainty Boundary via Shannon Entropy}
Let $\mathbf{x}$ represent the logit vector generated by an autoregressive language model over a vocabulary space $\mathcal{V}$, and let $P(w_i \mid \mathbf{x}) = \text{Softmax}(\mathbf{x} / T)$ be the conditional probability distribution parameterized by temperature $T > 0$.
\begin{assumption}
The conditional probability distribution is strictly continuous and supported on a non-empty, finite compact set $\mathcal{V}$.
\end{assumption}
\begin{theorem}
Minimizing the token uncertainty entropy $L_{\text{entropy}}$ strictly bounds the dynamic variance of the model's confidence, preventing catastrophic chaotic hallucinations caused by token distribution flattening.
\end{theorem}
\begin{proof}
We express the continuous target functional using Shannon Information Entropy formulation:
\begin{equation}
L_{\text{entropy}} = -\sum_{i=1}^{|\mathcal{V}|} P(w_i \mid \mathbf{x}) \log \left( P(w_i \mid \mathbf{x}) + \epsilon \right)
\end{equation}
where $\epsilon > 0$ is an infinitesimal stabilizing constant safeguarding against fractional zero singularity boundaries. Since $P(w_i \mid \mathbf{x}) \in [0, 1]$ and $\sum P(w_i \mid \mathbf{x}) = 1$, the loss functional possesses a strict global minimum $L_{\text{entropy}} = 0$ realized if and only if there exists a deterministic index $k$ such that $P(w_k \mid \mathbf{x}) = 1$ and $P(w_i \mid \mathbf{x}) = 0$ for all $i \neq k$. 
As $L_{\text{entropy}} \to 0$, the probability mass concentrates entirely on a singular semantic trajectory. This eliminates stochastic noise propagation through autoregressive token generation loops, bounded by the uniform continuity guaranteed under Assumption 1.
\end{proof}
\section{Theorem 2: Structural Geometric Contextual Alignment}
Let $\mathbf{H} \in \mathbb{R}^{B \times S \times D}$ denote the hidden state tensors representing the model's internal representation, and let $\mathbf{C} \in \mathbb{R}^{B \times K \times D}$ denote the true ground-truth contextual embedding space.
\begin{assumption}
The contextual mapping operator is globally Lipschitz continuous, and the embedding spaces share identical mathematical tracking dimensions $D$.
\end{assumption}
\begin{theorem}
Under Assumption 2, minimizing the localized geometric distance trace metric $\text{Tr}(\mathbf{D}^T \mathbf{D})$ ensures that the hidden representations remain strictly isomorphic to the target truth manifold, preventing semantic fabrication.
\end{theorem}
\begin{proof}
Let $h_i \in \mathbb{R}^D$ and $c_j \in \mathbb{R}^D$ represent the respective row vectors of $\mathbf{H}$ and $\mathbf{C}$. We formulate the exact pairwise coordinate tracking metric under a non-linear geometric projection scheme:
\begin{equation}
L_{\text{projection}} = \frac{1}{B \cdot S} \sum_{i=1}^{B \cdot S} \min_{j} \Vert h_i - c_j \Vert_2^2
\end{equation}
By Assumption 2, since the target embedding space $\mathbf{C}$ forms a closed, bounded, and compact manifold within $\mathbb{R}^D$, the localized minimization operator is strictly well-defined, and the derivative $\nabla_h L_{\text{projection}}$ is Lipschitz bounded. The condition $L_{\text{projection}} \to 0$ mandates that for every generated hidden coordinate $h_i$, there exists an exact semantic anchor $c_j$ within the true contextual matrix such that:
\begin{equation}
h_i \equiv c_j
\end{equation}
This establishes an exact topological embedding of the model's generation trajectory into the canonical truth manifold, mathematically eliminating the logical possibility of structural context divergence (hallucination).
\end{proof}
\section{Theorem 3: Multi-Loss Parity Convergence}
The complete operational objective integrates the optimization targets via a scale factor: $L_{\text{total}} = L_{\text{entropy}} + \alpha L_{\text{projection}}$.
\begin{theorem}
The fused optimization functional achieves continuous global stability and strict parameter convergence under localized convexity constraints.
\end{theorem}
\begin{proof}
Because $L_{\text{entropy}}$ is bounded from below by zero ($L_{\text{entropy}} \ge 0$) and $L_{\text{projection}}$ is bounded by the distance identity metric ($L_{\text{projection}} \ge 0$), the total combined loss functional is strictly bounded from below:
\begin{equation}
L_{\text{total}} \ge 0
\end{equation}
Taking the partial derivatives demonstrates a mutual dampening feedback cycle. As the optimization algorithm processes the continuous parameter path, the combined gradients act as a self-correcting negative feedback mechanism. This drives the hidden state architecture smoothly into the unique pre-defined global minimum configuration without gradient collapse or numerical explosion, establishing comprehensive topological stability.
\end{proof}
